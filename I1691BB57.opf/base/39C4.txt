བམ་པོ ༼༢༽ དང་པོ་༑༑་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ་༑༑
འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན་༑་བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ་ན་༑་བྱ་རྒོད་ཀྱི་ཕུང་པོའི ༼༣༽ རིལ་བཞུགས་ཏེ་༑་དགེ་སློང་ཁྲི་ཉིས་སྟོང་གི་དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་ཐམས་ཅད་ཀྱང་དགྲ་བཅོམ་པ་༑་ཟག་པ་ཟད་པ་༑་ཉོན་མོངས་པ་མེད་པ་༑་དབང་དང་ལྡན་པར་གྱུར་པ་༑་སེམས་ཤིན ༼༤༽ ཏུ་རྣམ་པར་གྲོལ་བ་༑་ཤེས་རབ་ཤིན་ཏུ་རྣམ་པར་གྲོལ་བ་༑་ཅང་ཤེས་པ་༑་གླང་པོ་ཆེན་པོ་༑་བྱ་བ་བྱས་པ་༑་བྱེད་པ་བྱས་པ་༑་ཁུར་བོར་བ་༑་བདག་གི་དོན་རྗེས་སུ་ཐོབ་པ་༑་སྲིད་པར་ཀུན་ཏུ ༼༥༽ སྦྱོར་བ་ཡོངས་སུ་ཟད་པ་༑་ཡང་དག་པའི་ཤེས་པས་སེམས་ཤིན་ཏུ་རྣམ་པར་གྲོལ་བ་༑་སེམས་ཀྱི་དབང་ཐམས་ཅད་ཀྱི་དམ་པའི་ཕ་རོལ་ཏུ་སོན་པ་༑་མངོན་པར་ཤེས་པས་མངོན་པར་ཤེས་པའི ༼༢ཨ༡༽ ཉན་ཐོས་ཆེན་པོ་ཤ་སྟག་ལའདི་ལྟ་སྟེ་༑་ཚེ་དང་ལྡན་པ་ཀུན་ཤེས་ཀཽཎཌིཉ་དང་༑་ཚེ་དང་ལྡན་པ་རྟ་ཐུལ་དང་༑་ཚེ་དང་ལྡན་པ་རླངས་པ་དང་༑་ཚེ་དང་ལྡམ་པ་མིང་ཆེན་དང་༑་ཚེ་དང་ལྡན་པ་བཟང་པོ་དང་༑་ཚེ་དང་ལྡན་པ་འོད་སྲུང་ཆེན་པོ་དང་༑་ཚེ་དང ༼༢༽ ལྡན་པ་ལྟང་རྒྱས་འོད་སྲུང་དང་༑་ཚེ་དང་ལྡན་པ་ཆུ་ཀླུང་འོད་སྲུང་དང་༑་ཚེ་དང་ལྡན་པ་ག་ཡ་འོད་སྲུང་དང་༑་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་དང་༑་ཚེ་དང་ལྡན་པ་མཽད་གལ་གྱི་བུ་ཆེན་པོ་དང་༑་ཚེ་དང་ལྡན་པ་ཀཱཊྱཱའི་བུ་ཆེན་པོ་དང་༑་ཚེ་དང་ལྡན་པ་མ་འགགས་པ་དང་༑་ཚེ་དང་ལྡན་པ ༼༣༽ ནམ་གྲུ་དང་༑་ཚེ་དང་ལྡན་པ་ཀ་པི་ན་དང་༑་ཚེ་དང་ལྡན་པ་པ་ལང་བ་དག་དང་༑་ཚེ་དང་ལྡན་པ་པི་ལིན་ངའི་བུ་དང་༑་ཚེ་དང་ལྡན་པ་བཀྐུ་ལ་དང་༑་ཚེ་དང་ལྡན་པ་གསུས་པོ་ཆེ་དང་༑་ཚེ་དང་ལྡན་པ་བྷ་ར་དྭ་ཛ་དང་༑་ཆེ་དང་ལྡན་པ་དགའ་བོ་དང་༑་ཚེ་དང་ལྡན་པ་ཉེ་དགའ་དང་༑ ༼༤༽ ཚེ་དང་ལྡན་པ་མཛེས་དགའ་དང་༑་ཚེ་དང་ལྡན་པ་བྱམས་མའི་བུ་གང་པོ་དང་༑་ཚེ་དང་ལྡན་པ་རབ་འབྱོར་དང་༑་ཚེ་དང་ལྡན་པ་སྒྲ་གཅན་ཟིན་དང་༑དེ་དག་དང་ཉན་ཐོས་ཆེན་པོ་གཞན་དག་དང་༑་སློབ་པ་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་དང་༑གཞན་ཡང་སློབ་པ་དང་མི ༼༥༽ སློབ་པའི་དགེ་སློང་ཉིས་སྟོང་དང་༑སྐྱེ་དགུའི་བདག་མོ་ཆེན་མོ་ལ་སོགས་པ་དགེ་སློང་མ་དྲུག་སྟོང་དང་༑་སྒྲ་གཅན་ཟིན་གྱི་མ་དགེ་སློང་མ་གྲངས་འཛིན་འཁོར་དང་བཅས་པ་དང་ཐབས་ཅིག་གོ་༑༑བྱང་ཆུབ་སེམས་དཔའ་བརྒྱད་ཁྲི་ཐམས་ཅད་ཀྱང་བླ་ན་མེད་པ་ཡང ༼༦༽ དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལས་ཕྱིར་མི་ལྡོག་པ་༑་གཟུངས་ཐོབ་པ་༑་སྤོབས་པ་ཆེན་པོ་ལ་གནས་པ་༑་ཕྱིར་མི་ལྡོག་པའི་ཆོས་ཀྱི་འཁོར་ལོ་སྐོར་བ་༑་སངས་རྒྱས་བརྒྱ་སྟོང་མང་པོ་ལ་བསྙེན་བཀུར་བྱས་པ་༑་སངས་རྒྱས་བརྒྱ་སྟོང་མང་པོ་ལ་དགེ་བའི་རྩ་བ་བསྐྱེད ༼༢བ༡༽ པ༑་སངས་རྒྱས་བརྒྱ་སྟོང་མང་པོས་ཡང་དག་བར་བསྟོད་པ་༑་ལུས་དང་དག་དང་སེམས་བྱམས་པས་ཡོངས་སུ་བསྒོས་པ་༑་དེ་བཞིན་གཤེགས་པའི་ཡེ་ཤེས་འཛུད་པ་ལ་མཁས་པ་༑་ཤེས་རབ་ཆེན་པོ་༑༑་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་བྱིན་བརྟོགས་པར་ཁོང་དུ་ཆུད་པ་༑་འཇིག་རྟེན་གྱི ༼༢༽ ཁམས་བརྒྱ་སྟོང་མང་པོར་རྣམ་པར་གྲགས་པ་༑་སྲོག་ཆགས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་མང་པོ་ཡང་དག་པར་སྒྲོལ་བ་ཤ་སྟག་ལ་འདི་ལྟ་སྟེ་༑བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག ༼༣༽ དང་༑་བྱང་ཆུབ་སེམས་དཔའ་མཐུ་ཆེན་ཐོབ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་དོན་ཐམས་ཅད་ཅེས་བྱ་བ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་རྟག་ཏུ་བརྩོན་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་བརྩོན་པ་མི་གཏོང་བ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་ལག་ན་རིན་ཆེན་དང་༑་བྱང་ཆུབ་སེམས་དཔའ ༼༤༽ སྨན་གྱི་རྒྱལ་པོ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་སྨན་ཡག་དག་འཕགས་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་བཀོད་པའི་རྒྱལ་པོ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་རབ་ཏུ་སྦྱིན་དཔའ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་རིན་ཆེན་ཟླ་བ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་རིན་ཆེན་འོད་དང་༑་བྱང་ཆུབ ༼༥༽ སེམས་དཔའ་ཟླ་གང་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་གནོན་པ་ཆེན་པོ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་མཐའ་ཡས་གནོན་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་འཇིག་རྟེན་གསུམ་གནོན་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་སྤོབས་པ་ཆེན་པོ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་རྟག་པར་རྒྱུན་ཏུ་བརྩོན ༼༦༽ དང་༑་བྱང་ཆུབ་སེམས་དཔའ་ས་འཛིན་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་བློ་གྲོས་མི་ཟད་པ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་པད་མའི་དཔལ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་སྐར་མའི་རྒྱལ་པོ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་བྱམས་པ་དང་༑་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ ༼༧༽ ཆེན་པོ་སེང་གེ་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑བཟང་སྐྱོང་ལ་སོགས་པ་སྐྱེས་བུ་དམ་པ་བཅུ་དྲུག་ལའདི་ལྟ་སྟེ་༑་བཟང་སྐྱོང་དང་༑་དཀོན་མཆོག་འབྱུང་གནས་དང་༑་དེང་དཔོན་བཟང་པོ་དང་༑་མེས་བྱིན་དང་༑་ཕུག་སྤས་དང་༑་ཆུས་བྱིན་དང་༑་དབང་པོས་བྱིན་དང་༑་བླ་མའི་བློ ༼༣ཨ༡༽ ༽]་༑་གྲོས་དང་༑་ཁྱད་པར་བློ་གྲོས་དང་༑་འཕེལ་བའི་བློ་གྲོས་དང་༑་མཐོང་བ་དོན་ཡོད་དང་༑་ཤིན་ཏུ་ཡང་དག་ཞུགས་དང་༑་རབ་ཀྱི་རྩལ་གྱིས་རྣམ་པར་གནོན་པ་དང་༑་དཔེ་མེད་བློ་གྲོས་དང་༑་ཉི་མའི་སྙིང་པོ་དང་༑་ས་འཛིན་དག་དང་ཡང་ཐབས་ཅིག་སྟེ་༑་དེ་དག་ལ་སོགས ༼༢༽ པ་བྱང་ཆུབ་སེམས་དཔའ་བརྒྱད་ཁྲི་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑ལྷ་རྣམས་ཀྱི་དབང་པོ་བརྒྱ་བྱིན་དང་༑་འཁོར་ལྷའི་བུ་ཉི་ཁྲིའདི་ལྟ་སྟེ་༑་ལྷའི་བུ་ཟླ་བ་དང་༑་ལྷའི་བུ་ཉི་མ་དང་༑་ཀུན་དུ་དྲི་དང་༑་རིན་ཆེན་འོད་དང་༑་ལྷའི་བུ་སྣང་འོད་དང་༑་དེ་དག་ལ་སོགས་པ་ལྷའི་བུ་ཉིད་ཁྲི་དང་ཡང་ཐབས ༼༣༽ ཅིག་གོ་༑༑རྒྱལ་པོ་ཆེན་པོ་བཞི་དང་༑་འཁོར་ལྷའི་བུ་སུམ་ཁྲིའདི་ལྟ་སྟེ་༑་རྒྱལ་པོ་ཆེན་པོ་འཕགས་སྐྱས་པོ་དང་༑་རྒྱལ་པོ་ཆེན་པོ་མིག་མི་བཟང་དང་༑་རྒྱལ་པོ་ཆེན་པོ་ཡུལ་འཁོར་སྲུང་དང་༑་རྒྱལ་པོ་ཆེན་པོ་རྣམ་ཐོས་ཀྱི་བུ་དང་༑ལྷའི་བ་དབང་ཕྱུག་དང་༑་ལྷའི་བུ་དབང་ཕྱུག་ཆེན་པོ་དང་༑ ༼༤༽ འཁོར་ལྷའི་བུ་སུམ་ཁྲི་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑མི་མཇེད་ཀྱི་བདག་པོ་ཚངས་པ་དང་༑་འཁོར་ཚངས་རིས་ཀྱི་ལྷའི་བུ་ཁྲི་ཉིས་སྟོངའདི་ལྟ་སྟེ་༑་ཚངས་པ་གཙུག་ཕུད་ཅན་དང་༑་ཚངས་པ་སྐར་མའི་འོད་དང་༑དེ་དག་ལ་སོགས་པ་ཚངས་རིས་ཀྱི་ལྷའི་བུ་ཁྲི་ཉིས་སྟོང་དང་ཡང་ཐབས་ཅིག་གོ་༑༑ཀླུའི་རྒྱལ ༼༥༽ པོ་བརྒྱདའདི་ལྟ་སྟེ་༑་ཀླུའི་རྒྱལ་པོ་དགའ་བོ་དང་༑་ཉེ་དགའ་དང་༑་རྒྱ་མཚོ་དང་༑་ནོར་རྒྱས་ཀྱི་བུ་དང་༑་འཇོག་པོ་དང་༑་གཟི་ཅན་དང་༑་མ་དྲོས་པ་དང་༑་ཀླུའི་རྒྱལ་པོ་ཨུད་པ་ལ་དང་༑་འཁོར་ཀླུ་བྱེ་བ་བརྒྱ་སྟོང་མང་པོ་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑མི་འམ་ཅིའི་རྒྱལ་པོ་བཞིའདི་ལྟ་སྟེ་༑ ༼༦༽ མིའམ་ཅིའི་རྒྱལ་པོ་ལྗོན་པ་དང་༑་མི་འམ་ཅིའི་རྒྱལ་པོ་ཆོས་ཆེན་དང་༑་མི་འམ་ཅིའི་རྒྱལ་པོ་ཆོས་བཟང་དང་༑་མི་འམ་ཅིའི་རྒྱལ་པོ་ཆོས་འཛིན་དང་༑་འཁོར་མི་འམ་ཅི་བྱེ་བ་བརྒྱ་སྟོང་མང་པོ་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑དྲི་ཟའི་རིས་ཀྱི་ལྷའི་བུ་བཞིའདི་ལྟ་སྟེ་༑་དྲི་ཟ་ཡིད་དུ་འོང་བ་དང་༑ ༼༧༽ ཡིད་འོང་དབྱངས་དང་༑་འཇམ་སྙན་དང་༑་དྲི་ཟ་དབྱངས་སྙན་དང་༑་འཁོར་དྲི་ཟ་བྱེ་བ་བརྒྱ་སྟོང་མང་པོ་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑ལྷ་མ་ཡིན་གྱི་དབང་པོ་བཞིའདི་ལྟ་སྟེ་༑་ལྷ་མ་ཡིན་གྱི་དབང་པོ་སྟོབས་ཅན་དང་༑་ཕྲག་རྩུབ་དང་༑་ཐགས་བཟངས་རིས་དང་༑་ལྷ་མ ༼༣བ༡༽ ཡིན་གྱི་དབང་པོ་སྒྲ་གཅན་དང་༑་འཁོར་ལྷ་མ་ཡིན་བྱེ་བ་བརྒྱ་སྟོང་མང་པོ་དག་དང་ཡང་ཐབས་ཅིག་གོ་༑༑ནམ་མཁའ་ལྡིང་གི་དབང་པོ་བཞིའདི་ལྟ་སྟེ་༑་ནམ་མཁའ་ལྡིང་གི་དབང་པོ་གཟི་བརྗིད་ཆེན་པོ་དང་༑་ལུས་ཆེན་དང་༑་རྫོགས་ཆེན་དང་༑་ནམ་མཁའ་ལྡིང་གི་དབང་པོ་འཕྲུལ དེའི་ཚེ་ན་བཅོམ་ལྡན་འདས་འཁོར་བཞི་པོ་དེ་དག་གིས་ཡོངས་སུ་བསྐོར་ཅིང་མདུན་གྱིས་བལྟས་ཏེ་༑ ༼༣༽ བསྟི་སྟང་དུ་བྱས་༑་བཙུན་པར་བྱས་༑་རི་མོར་བྱས་༑་མཆོད་པར་བྱས་༑་རྗེད་པར་བྱས་༑་གསོལ་བ་བྱས་ཏེ་༑་ངེས་པར་བསྟན་པ་ཆེན་པོ་ཞེས་བྱ་བའི་ཆོས་ཀྱི་རྣམ་གྲངས་མདོ་སྡེ་ཤིན་ཏུ་རྒྱས་པ་ཆེན་པོ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་གདམས་པ་༑་སངས་རྒྱས་ཐམས་ཅད་ཀྱིས ༼༤༽ ཡོངས་སུ་བཟུང་བ་བཤད་པར་བཞེད་དེ་ཆོས་ཀྱི་གདན་དེ་ཉིད་ལ་སྐྱིལ་མོ་ཀྲུང་བཅས་ནས་༑་མཐའ་ཡས་ངེས་པར་བསྟན་པའི་གནས་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་ཏེ་སྐུ་མི་ག་ཡོ་ཞིང་སེམས་མི་ག་ཡོ་བར་གྱུར་ནས་བཞུགས་སོ་༑༑བཅོམ་ལྡན་འདས་སྙོམས་པར་ཞུགས ༼༥༽ མ་ཐག་ཏུ་༑་བཅོམ་ལྡན་འདས་ལ་ལྷའི་མེ་ཏོག་མན་དཱ་ར་དང་༑་མན་དཱར་ཆེན་པོ་དང་༑་མཉྫུ་ཤ་ཀ་ཆེན་པོ་རྣམས་ཀྱི་མེ་ཏོག་གི་ཆར་ཆེན་པོ་མངོན་པར་བབ་སྟེ་༑་བཅོམ་ལྡན་འདས་དང་འཁོར་བཞི་པ་དི་དག་ལ་འཐོར་རོ་༑༑ཐམས་ཅད་དང་ལྡན་པའི་སངས་རྒྱས་ཀྱི་ཞིང་ཡང ༼༦༽ རྣམ་པ་དྲུག་ཏུ་ག་ཡོས་པར་གྱུར་ཏེ་༑་འགུལ་ཀུན་ཏུ་རབ་ཏུ་འགུལ་༑་ལྡེག་ཀུན་ཏུ་རབ་ཏུ་ལྡེག་༑་འཁྲུགས་ཀུན་ཏུ་རབ་ཏུ་འཁྲུགས་སོ་༑༑དེའི་ཚེ་དགེ་སློང་དང་༑་དགེ་སློང་མ་དང་༑་དགེ་བསྙེན་དང་༑་དགེ་བསྙེན་མ་དང་༑་ལྷ་དང་༑་ཀླུ་དང་༑་གནོད་སྦྱིན་དང་༑་དྲི་ཟ་དང་༑་ལྷ་མ ༼༧༽ ཡིན་དང་༑་ནམ་མཁའ་ལྡིང་དང་༑་མི་འམ་ཅི་དང་༑་ལྟོ་འཕྱེ་ཆེན་པོ་དང་༑་མི་དང་༑་མི་མ་ཡིན་པ་རྣམས་དང་༑་རྒྱལ་ཕྲན་དང་༑་ལྟོབས་ཀྱི་འཁོར་ལོས་སྒྱུར་བ་དང་༑་གླིང་བཞིའི་འཁོར་ལོས་སྒྱུར་བ་གང་དག་འཁོར་དེར་འདུས་ནསའཁོད་པར་གྱུར་པ་འཁོར་དང་བཅས་པ་དེ་དག ༼༤ཨ༡༽ ༑་ཐམས་ཅད་བཅོམ་ལྡན་འདས་ལ་ལྟ་ཞིང་འཁོད་དེ་ངོ་མཚར་དུ་གྱུར་༑་རྨད་དུ་གྱུར་༑་ཤིན་ཏུ་དགའ་བར་གྱུར་ཏོ་༑༑
དེ་ནས་ངེའི་ཚེ་བཅོམ་ལྡན་འདས་ཀྱི་སྨིན་མཚམས་ཀྱི་མཛོད་སྤུ་ནས་འོད་ཟེར་གཅིག་བྱུང་ནས་༑དེ་ཤར་ཕྱོགས་སུ་སངས་རྒྱས་ཀྱི་ཞིང་ཁྲི་བརྒྱད་སྟོང་དུ་སོང་སྟེ་༑༼༢༽ སངས་རྒྱས་ཀྱི་ཞིང་དེ་དག་ཐམས་ཅད་མནར་མེད་པའི་སེམས་ཅན་དམྱལ་བ་ཆེན་པོ་ཡན་ཆད་སྲིད་པའི་རྩེ་མོ་མན་ཆད་འོད་ཟེར་དེའི་འོད་ཀྱིས་ཤིན་ཏུ་ཁྱབ་པར་སྣང་ངོ་༑༑སངས་རྒྱས་ཀྱི་ཞིང་དེ་དག་གི་འགྲོ་བ་དྲུག་པོ་དེ་དག་ན་སེམས་ཅན་གང་ཡོད་པ་དེ་དག་ཐམས་ཅད་མ་ལུས་པར་སྣང་ངོ་༑༑སངས ༼༣༽ རྒྱས་ཀྱི་ཞིང་དེ་དག་ན་སངས་རྒྱས་བཅོམ་ལྡན་འདས་གང་དག་བཞུགས་ཤིང་འཚོག་ཞེས་པ་དེ་དག་ཐམས་ཅད་ཀྱང་སྣང་ངོ་༑༑སངས་རྒྱས་བཅོམ་ལྡན་འདས་དེ་དག་ཅོས་གང་འཆད་པ་དེ་དག་ཐམས་ཅད་ཀྱང་མ་ལུས་པར་གྲག་གོ་༑༑སངས་རྒྱས་ཀྱི་ཞིང་དེ་དག་ན་དགེ་སློང་དང་༑་དགེ་སློང་མ་དང་༑ ༼༤༽ དགེ་བསྙེན་པ་༑་དགེ་བསྙེན་མ་དང་༑་རྣལ་འབྱོར་ཅན་དང་༑་རྣལ་འབྱོར་སྤྱོད་པ་གང་དག་འབྲས་བུ་ཐོབ་པ་དང་༑་འབྲས་བུ་མ་ཐོབ་པ་དེ་དག་ཐམས་ཅད་ཀྱང་སྣང་ངོ་༑༑དམིགས་པ ༼༥༽ དང་༑་མོས་པ་རྣམ་པ་སྣ་ཚོགས་དུ་མའི་རྒྱུའི་ཐབས་ལ་མཁས་པ་དག་གིས་བྱང་ཆུབ་སེམས་དཔའི་སྤྱད་པ་སྤྱོད་པ་དེ་དག་ཐམས་ཅད་ཀྱང་སྣང་ངོ་༑༑སངས་རྒྱས་ཀྱི་ཞིང་དེ་དག་ན་༑་སངས་རྒྱས་བཅོམ་ལྡན་འདས་གང་དག་ཡོངས་སུ་མྱ་ངན་ལས་འདའ་བ་དེ་དག་ཐམས་ཅད་ཀྱང་སྣང་ངོ་༑༑སངས ༼༦༽ རྒྱས་ཀྱི་ཞིང་དེ་དག་ན་༑་སངས་རྒྱས་བཅོམ་ལྡན་འདས་གང་དག་ཡོངས་སུ་མྱ་ངན་ལས་འདས་པ་རྣམས་ཀྱི་རིང་བསྲེལ་གྱི་མཆོད་རྟེན་རིན་པོ་ཆེ་ལས་བྱས་པ་དེ་དག་ཐམས་ཅད་ཀྱང་སྣང་ངོ་༑༑
དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པ་འདི་སྙམ་དུ་སེམས་ཏེ་༑་ཀྱེ་མ ༼༧༽ དེ་བཞིན་གཤེགས་པས་ལྟས་ཆོ་འཕྲུལ་ཆེན་པོ་བསྟན་པ་འདི་འདིར་ཅིའི་རྒྱུར་འགྱུར་ཞིག་གོ་༑༑ཅིའི་ཕྱིར་བཅོམ་ལྡན་འདས་ཀྱིས་ལྟས་ཆོ་འཕྲུལ་ཆེན་པོ་འདི་ལྟ་བུ་འདི་བསྟན་ཏེ་༑འདས་ཀྱང་ཏིང་ངེ་འཛིན་དེ་ལ་སྙོམས་པར་ཞུགས་ཤིང་རྫུ་འཕྲུལ་གྱི་ཆོ་འཕྲུལ་ཆེན་པོ་ངོ་མཚར་རྨད ༼༤བ༡༽ དུ་བྱུང་བ་ཆེན་པོ་བསམ་གྱིས་མི་ཁྱབ་པ་འདི་ལྟ་བུ་འདི་དག་ཀྱང་སྣང་བར་འགྱུར་བའདིའི་དོན་བདག་གིས་སུ་ལ་དྲིའདི་ན་དོན་འདིའི་ལན་ལྡོན་ནུས་པ་ནི་སུ་ཡོད་སྙམ་མོ་༑༑དེ་ཡང་འདི་སྙམ་དུ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་འདི་སྔོན་གྱི་རྒྱལ་བ་རྣམས་ལ་ལྷག་པར་བྱ་བ་བྱས་པ་༑་དགེ་བའི་རྩ ༼༢༽ བ་བསྐྱེད་པ་༑་སངས་རྒྱས་མང་པོ་ལ་བསྙེན་བཀུར་བྱས་པ་སྟེ་༑འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་འདིས་སྔོན་ཆད་ཀྱང་སྔོན་གྱི་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ཀྱི་ལྟས་འདི་ལྟ་བུ་བྱུང་བ་རྣམས་མཐོང་ཞིང་སྔོན་ཆད་ཆོས་ཆེན་པོའི ༼༣༽ འབེལ་བའི་གཏམ་དག་ཀྱང་མྱོང་བས་ན་༑བདག་གིས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་དེའི་དོན་དྲིའོ་སྙམ་དུ་སེམས་སོ་༑༑

དེ་ནས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་བྱམས་པ་དང་༑་ཐམས་ཅད་དང་ལྡན་པའི་བྱང་ཆུབ་སེམས་དཔའི་ཚོགས་དེ་ལ་སྨྲས་པ་༑རིགས་ཀྱི་བུ་དག་འདི་ནི་དེ་བཞིན་གཤེགས་པ་ཆོས་ཆེན་པོ་བསྒྲགས ༼༣༽ པའི་འབེལ་བའི་གཏམ་མཛད་པར་དགོངས་པའོ་༑༑་རིགས་ཀྱི་བུ་དག་འདི་ནི་དེ་བཞིན་གཤེགས་པས་ཆོས་ཀྱི་ཆར་ཆེན་པོ་མངོན་པར་དབབ་པ་དང་༑་ཆོས་ཀྱི་རྔ་ཆེན་པོ་རབ་ཏུ་བརྡུང་བ་དང་༑༑་ཆོས་ཀྱི་རྒྱལ་མཚན་ཆེན་པོ་རབ་ཏུ་བསྒྲེང་བ་དང་༑་ཆོས་ཀྱི་སྒྲོན་མ་ཆེན་པོ་རབ་ཏུ་འབར་བར ༼༤༽ མཛད་པ་དང་༑་ཆོས་ཀྱི་དུང་ཆེན་པོ་རབ་ཏུ་འབུད་པ་དང་༑་ཆོས་ཀྱི་རྔ་བོ་ཆེ་ཆེན་པོ་རབ་ཏུ་བརྡུང་བ་དང་༑་ཆོས་བསྟན་པ་ཆེན་པོ་མཛད་པར་དགོངས་པ་སྟེ་༑རིགས་ཀྱི་བུ་དག་བདག་ཇི་ལྟར་སྤོབས་པ་དང་༑་ཇི་ལྟར་སྔོན་གྱི་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི ༼༥༽ སངས་རྒྱས་རྣམས་ཀྱི་སྔ་ལྟས་འདི་ལྟ་བུ་འདི་བདག་གིས་མཐོང་སྟེ་༑་སྔོན་གྱི་དེ་བཞིན་གཤེགས་པ་དེ་དག་གིས་ཀྱང་༑་འོད་ཟེར་གྱི་སྣང་བ་འདི་བཏང་བས་༑བདག་ནི་འདི་ལྟར་དེ་བཞིན་གཤེགས་པ་ཡང་ཆོས་ཆེན་པོ་བསྒྲག་པའི་འཕེལ་བའི་གཏམ་མཛད་པར་བཞེད་པ་དང་༑ ༼༦༽ ཆོས་ཆེན་པོ་བསྒྲགས་པ་ཐོས་པར་མཛད་པ་བཞེད་པས་འདི་ལྟ་བུའི་སྔ་ལྟས་འདི་མཛད་པ་སྙམ་བྱེད་དོ་༑༑དེ་ཅིའི་ཕྱིར་ཞེ་ན་༑་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་འཇིག་རྟེན་ཐམས་ཅད་དང་མི་འཐུན་པའི་ཆོས་ཀྱི་རྣམ་གྲངས་བསྒྲག་པར་བཞེད་པས་སྔ ༼༧༽ ལྟས་ཆོ་འཕྲུལ་ཆེན་པོ་འོད་ཟེར་རབ་ཏུ་བཏང་བའི་སྣང་བ་འདི་ལྟ་བུ་འདི་སྟོན་ཏོ་༑༑
རིགས་ཀྱི་བུ་དག་བདག་གིས་དྲན་པ་འདས་པའི་དུས་ན་བསྐལ་པ་གྲངས་མེད་པའི་ཡང་ཆེས་གྲངས་མེད་པ་ཡངས་པ་དཔག་ཏུ་མེད་པ་བསམ་གྱིས་མི་ཁྱབ་པ་དཔག་གིས་མི་ལང་བ་ཚད་མེད་པ་དེའི་ཕ་རོལ ༼༨ཨ༡༽ གྱི་ཡང་ཆེས་ཕ་རོལ་ཏུ་གྱུར་པ་དེའི་ཚེ་དེའི་དུས་ན་༑་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རིག་པ་དང་ཞབས་སུ་ལྡན་པ་བདེ་བར་གཤེགས་པ་འཇིག་རྟེན་མཁྱེན་པ་སྐྱེས་བུ་འདུལ་བའི་ཁ་ལོ་སྒྱུར་བ་བླ་ན་མེད་པ་ལྷ་དང་མི་རྣམས ༼༢༽ ཀྱི་སྟོན་པ་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ཉི་ཟླ་སྒྲོན་མ་ཞེས་བྱ་བ་འཇིག་རྟེན་དུ་བྱུང་སྟེ་༑དེས་ཆོས་བསྟན་པ་ཚངས་པར་སྤྱོད་པ་ཐོག་མར་དགེ་བ་བར་དུ་དགེ་བ་ཐ་མར་དགེ་བདོན་བཟང་པོ་ཚིག་འབྲུ་བཟང་པོ་མ་འདྲིས་པ་ཡོངས་སུ་རྫོགས་པ་ཡོངས་སུ་དག་པ་ཡོངས་སུ་བྱང་བ་ཡང་དག་པར ༼༣༽ སྟོན་ཏེའདི་ལྟ་སྟེ་༑་ཉན་ཐོས་རྣམས་ལ་ནི་སྐྱ་བ་དང་རྒ་བ་དང་༑་ནབ་དང་འཆི་བ་དང་༑་མྱ་ངན་དང་སྨྲེ་སྔགས་འདོན་པ་དང་༑་སྡུག་བསྔལ་བ་དང་༑་ཡིད་མི་བདེ་བ་དང་འཁྲུག་པ་ལས་ཤིན་ཏུ་བཟླ་བའི་ཕྱིར་༑་མྱ་ངན་ལས་འདས་པའི་མཐར་ཐུག་པ་འཕགས་པའི་བདེན་པ ༼༤༽ བཞི་དང་ལྡན་པ་༑་རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བ་ལ་འཇུག་པའི་ཆོས་སྟོན་ཏོ་༑༑བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ལ་ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ལས་བརྩམས་ཏེ་༑་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་དང་ལྡན་པ་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཀྱི་མཐར ༼༥༽ ཐུག་པའི་ཆོས་སྟོན་ཏོ་༑༑
རིགས་ཀྱི་བུ་དག་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཉི་ཟླ་སྒྲོན་མདའི་ཕ་རོལ་གྱི་ཡང་ཆེས་ཕ་རོལ་ན་༑་ཡང་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཉི་ཟླ་སྒྲོན་མ་ཞེས་བྱ་བ་ཞིག ༼༦༽ འཇིག་རྟེན་དུ་བྱུང་སྟེམ་ཕམ་པ་འདི་ལྟར་སྔ་རོལ་དང་སྔ་རོལ་བརྗོད་པའི་ཚུལ་འདིས་༑་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཉི་ཟླ་སྒྲོན་མ་ཞེས་བྱ་བར་མཚན་གཅིག་པ་རིགས་དང་གདུང་གཅིག་པ་ཤ་སྟག་སྟེ་༑་འདི་ལྟར་བྷར་དྭ་ཛའི་རིགས ༼༧༽ སུ་འཐུན་པ་དེ་བཞིན་གཤེགས་པ་ཉི་ཁྲི་བྱུང་ངོ་༑༑མ་ཕམ་པ་དེ་ལ་དེ་བཞིན་གཤེགས་པ་ཉི་ཁྲི་པོ་དེ་དག་ཐོག་མའི་དེ་བཞིན་གཤེགས་པ་ནས་ཐ་མའི་དེ་བཞིན་གཤེགས་པའི་བར་དུདེ་དག་གིས་ཆོས་བཤད་པ་ཡང་༑་ཚངས་པར་སྤྱོད་པ་ཐོག་མར་དགེ་བ་བར་དུ་དགེ་བ་ཐ་མར ༼༨བ༡༽ དགེ་བདོན་བཟང་པོ་ཚིག་འབྲུ་བཟང་པོ་མ་འདྲེས་པ་ཡོངས་སུ་རྫོགས་པ་ཡོངས་སུ་དག་པ་ཡོངས་སུ་བྱང་བ་ཡང་དག་པར་སྟོན་ཏེའདི་ལྟ་སྟེ་༑་ཉན་ཐོས་རྣམས་ལ་ནི་སྐྱེ་བ་དང་རྒ་བ་དང་ན་བ་དང་འཆི་བ་དང་༑་མྱ་ངན་དང་སྨྲེ་སྔགས་འདོན་པ་དང་སྡུག་བསྔལ་བ་དང་ཡིད་མི་བ་དེ་བ་དང་འཁྲུག་པ ༼༢༽ ལས་ཤིན་ཏུ་བཟླ་བའི་ཕྱིར་༑་མྱ་ངན་ལས་འདས་པའི་མཐར་ཐུག་པ་འཕགས་པའི་བདེན་པ་བཞི་དང་ལྡན་པ་རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བ་ལ་འཇུག་པའི་ཆོས་སྟོན་ཏོ་༑༑བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ལ་ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ ༼༣༽ ལས་བརྩམས་ཏེ་༑་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་དང་ལྡན་པ་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཀྱི་མཐར་ཐུག་པའི་ཆོས་སྟོན་ཏོ་༑༑

དེ་དག་གི་ཐ་མ་ནི་དེ་བཞིན་གཤེགས་པ་དགྲ ༼༦༽ བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་མར་མེ་མཛད་ཡིན་ནོ་༑༑སློབ་མ་བརྒྱད་བརྒྱ་པོ་དེ་དག་ལས་གཅིག་རྙེད་པ་ལ་རབ་ཏུ་གདུ་བ་བཀུར་སྟི་ལ་གདུ་པ་ཤེས་ཀྱི་ཁེལ་གདུ་བ་གྲགས་པ་འདོད་པས་༑་དེ་ལ་བསྟན་པའི་ཚིག་དང་༑་ཡི་གེ་མི་ཟིན་མི་ཆགས་ཏེདེའི་མིང་ཡང་གྲགས ༼༧༽ འདོད་ཅེས་བྱ་བར་གྱུར་ཏོ་༑༑དེས་ཀྱང་དགེ་བའི་རྩ་བ་དེས་སངས་རྒྱས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་མང་པོ་མཉེས་པར་བྱས་སོ་༑༑་མཉེས་པར་བྱས་ནས་བསྟི་སྟང་བྱས་༑་བཙུན་པར་བྱས་༑་རི་མོར་བྱས་༑༑་མཆོད་པ་དང་༑་བཀུར་བ་དང་༑་རིམ་གྲོ་བྱས་སོ་༑༑མ་ཕམ་བདེའི་ཚེ་དེའི ༼༡༠བ༡༽ དུས་ན་༑༑་བྱང་ཆུབ་སེམས་དཔའ་འོད་མཆོག་ཅེས་བྱ་བ་ཆོས་སྨྲ་བར་གྱུར་པ་དེ་༑་གཞན་ཞིག་ཡིན་པ་སྙམ་དུ་ཁྱོད་སོམ་ཉི་འམ་ཡིད་གཉིས་སམ་ཐེ་ཚོམ་ཟ་བར་གྱུར་ནདེ་ལྟར་མི་བལྟའོ་༑༑དེ་ཅིའི་བྱིར་ཞེ་ན་༑་དེ་ཅིའི་བྱིར་ཞེ་ན་༑་ཁོ་བོ་དེའི་ཚེ་དེའི་དུས་ན་བྱང་ཆུབ་སེམས་དཔའ་འོད་མཆོག་ཅེས ༼༢༽ བྱ་བ་ཆོས་སྨྲ་བར་གྱུར་ཏོ་༑༑གང་ཡང་བྱང་ཆུབ་སེམས་དཔའ་གྲགས་འདོད་ཅེས་བྱ་བ་ལེ་ལོ་ཅན་དུ་གྱུར་པ་དེ་ནི་༑་མ་ཕམ་པ་ཁྱོད་ཉིད་དེའི་ཚེ་དེའི་དུས་ན་༑་བྱང་ཆུབ་སེམས་དཔའ་གྲགས་འདོད་ཅེས་བྱ་བ་ལེ་ལོ་ཅན་དུ་གྱུར་པ་ཡིན་ནོ་༑༑མ་ཕམ་པ་དེ་ལྟར་ཁོ་བོས་རྣམ་གྲངས་འདིས་ན་བཅོམ ༼༣༽ ལྡན་འདས་ཀྱི་སྔ་ལྟས་འདི་ལྟ་བུའི་འོད་ཟེར་བཏང་བ་མཐོང་ནས་༑་བཅོམ་ལྡན་འདས་ཀྱང་དམ་པའི་ཆོས་པད་མ་དཀར་པོའི་ཆོས་ཀྱི་རྣམ་གྲངས་མདོ་སྡེ་ཤིན་ཏུ་རྒྱས་པ་ཆེན་པོ་དེ་བཤད་པར་བཞེད་པ་སྙམ་བྱེད་དོ་༑༑

༑༑་བམ་པོ་གཉིས་པ་༑༑ དེ ༼༥༽ ནས་བཅོམ་ལྡན་འདས་དྲན་པ་དང་ལྡན་ཞིང་མཁྱེན་བཞིན་དུ་ཏིང་ངེ་འཛིན་དེ་ལས་བཞེངས་སོ་༑༑་བཞེངས་ནས་ཚེ་དང་ལྡན་པ་ཤཱ་རི་དབ༹་ཏའི་བུ་ལ་བཀའ་སྩལ་པ་༑ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ཀྱིས་ཐུགས་སུ་ཆུད་པ་༑༑ ༼༦༽ སངས་རྒྱས་ཀྱི་ཡེ་ཤེས་ནི་ཟབ་པ་མཐོང་བར་དཀའ་བ་རྟོགས་པར་དཀའ་བ་སྟེ་༑༑་ཉན་ཐོས་དང་རང་སངས་རྒྱས་ཐམས་ཅད་ཀྱིས་ཤེས་པར་དཀའོ་༑༑དེ་ཅིའི་ཕྱིར་ཞེ་ན་༑་ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ནི་༑༑་སངས ༼༧༽ རྒྱས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་མང་པོ་ལ་བསྙེན་བཀུར་མཛད་མཛད་པ་༑་སངས་རྒྱས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་མང་པོ་ལ་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་སྤྱོད་པ་སྤྱད་སྤྱད་པ་༑་ཡུན་རིང་པོ་ནས་རྗེས་སུ་ཞུགས་པ་༑་བརྩོན་འགྲུས་མཛད་མཛད་པ་སྟེ་༑་ངོ་མཚར ༼༡༣ཨ༽ ༼༡༣ཨ༡༽ ༑་དང་རྨད་དུ་བྱུང་བའི་ཆོས་དང་ལྡན་ཞིང་ཞེས་པར་དཀའ་བའི་ཆོས་རྣམས་མཁྱེན་པའི་ཕྱིར་རོ་༑༑
ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ཀྱི་ལྡེམ་པོར་དགོངས་ཏེ་བཤད་པ་ནི་ཤེས་པར་དཀའོ་༑༑དེ་ཅིའི་ཕྱིར་ཞེ་ན་༑་རང ༼༢༽ གིས་རིག་པའི་ཆོས་རྣམས་ཐབས་མཁས་པ་དང་༑་ཡེ་ཤེས་མཐོང་བ་དང་༑་རྒྱུ་དང་༑་གཞི་དང་༑་དམིགས་པ་དང་༑་ངེས་པའི་ཚིག་དང་༑་རྣམ་པར་རིག་བྱེད་སྣ་ཚོགས་རྣམས་ཀྱིས་ཡང་དག་པར་སྟོན་ཏེ་༑་ཐབས་མཁས་པ་དེ་དང་དེ་དག་གིས་སེམས་ཅན་དེ་དང་དེ་ལ་ཆགས་པ་རྣམས་རབ ༼༣༽ ཏུ་འགྲེལ་པའི་ཕྱིར་རོ་༑༑ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ནི་ཐབས་མཁས་པ་ཆེན་པོ་དང་༑་ཡེ་ཤེས་མཐོང་བ་དམ་པའི་ཕ་རོལ་ཏུ་བྱོན་པའོ་༑༑ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི ༼༤༽ སངས་རྒྱས་རྣམས་ནི་༑་ཆགས་པ་མེད་ཅིང་ཐོགས་པ་མེད་པའི་ཡེ་ཤེས་མཐོང་བ་དང་༑་སྟོབས་དང་མི་འཇིགས་པ་དང་མ་འདྲེས་པ་དང་དབང་པོ་དང་སྟོབས་དང་བྱང་ཆུབ་ཀྱི་ཡན་ལག་དང་བསམ་གཏན་དང་༑་རྣམ་པར་ཐར་པ་དང་ཏིང་ངེ་འཛིན་དང་སྙོམས་པར་འཇུག་པ་དང་༑་རྨད ༼༥༽ དུ་བྱུང་བའི་ཆོས་དང་ལྡན་ཏེ་༑་ཆོས་རྣམ་པ་སྣ་ཚོགས་ཡང་དག་པར་སྟོན་པའོ་༑༑ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ནི་ངོ་མཚར་ཆེན་པོ་བརྙེས་པའོ་༑༑ཤཱ་རིའི་བུ་དེ་ཙམ་གྱིས་བཤད་པ་ཆོག་པར་གྱིས་ཤིག་༑་ཤཱ་རིའི་བུ་དེ ༼༦༽ བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ནི་ངོ་མཚར་མཆོག་བརྙེས་པ་སྟེ་༑ཤཱ་རིའི་བུ་དེ་བཞིན་གཤེགས་པས་ཆོས་གང་མཁྱེན་པའི་ཆོས་དེ་ཡང་དེ་བཞིན་གཤེགས་པས་དེ་བཞིན་གཤེགས་པ་ཉིད་ལ་འཆད་དོ་༑༑ཤཱ་རིའི་བུ་ཆོས་ཐམས་ཅད་ཀྱང ༼༧༽ དེ་བཞིན་གཤེགས་པ་ཉིད་སྟོན་ཏོ་༑༑ཆོས་ཐམས་ཅད་ཀྱང་དེ་བཞིན་གཤེགས་པ་ཉིད་ཀྱིས་མཁྱེན་ཏེ་ཆོས་དེ་དག་གང་ཡིན་པ་དང་༑་ཆོས་དེ་དག་ཇི་ལྟ་བུ་ཡིན་པ་དང་༑་ཆོས་དེ་དག་ཅི་འདྲ་བ་དང་༑་ཆོས་དེ་དག་གི་མཚན་ཉིད་གང་ཡིན་པ་དང་༑་ཆོས་དེ་དག་གི་ངོ་བོ་ཉིད་ཅི་ཡིན་པ་དང་༑ ༼༡༣བ༡༽ ཆོས་དེ་དག་གང་ཡིན་པ་དང་༑་ཇི་ལྟ་བུ་ཡིན་པ་དང་ཅི་འདྲ་བ་དང་༑་མཚན་ཉིད་གང་ཡིན་པ་དང་ངོ་བོ་ཉིད་ཅི་ཡིན་པ་ཡང་མཁྱེན་ཏོ་༑༑

དེ་ནས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་འཁོར་བཞི་པོ་དེ་ད་གསོམ་ཉི་དང་ཐེ་ཚོམ་ཟ་བར་རིག་སྟེ་སེམས་ཀྱི་རྣམ་པར་རྟོགས་པ་སེམས་ཀྱིས་ཤེས་ནས་༑་བདག་ཉིད་ཀྱང་ཆོས་ལ་ཐེ་ཚོམ་ཟ་བར་གྱུར་ཏེ་དེའི་ཚེ་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ་༑༑བཅོམ ༼༥༽ ལྡན་འདས་འདི་ལྟར་བཅོམ་ལྡན་འདས་ཀྱིས་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱི་ཐབས་མཁས་པ་མཁྱེན་པའི་ཆོས་བསྟན་པ་ཡང་དང་ཡང་བསྔགས་པ་མཛད་པ་དང་༑ཆོས་ཟབ་མོངས་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་སོ་ཞེས་བགྱི་བ་དང་༑ལྡེམ་པོར་དགོངས་ཏེ་བཤད་པ ༼༦༽ ཡང་ཤེས་པར་དཀའོ་ཞེས་ཡང་དང་ཡང་གསུངས་པའི་རྒྱུ་གང་ལགས་རྐྱེན་གང་ལགས་༑༑བདག་གིས་བཅོམ་ལྡན་འདས་ལས་ཆོས་ཀྱེ་རྣམ་གྲངས་འདི་ལྟ་བུ་སྔོན་ཆད་མཐོས་ཏེ་༑བཅོམ་ལྡན་འདས་འཁོར་བཞི་པོ་འདི་དག་ཀྱང་ཐེ་ཚོམ་དང་སོམ་ཉི་འཚལ་ན་༑༑དེ་བཞིན ༼༧༽ གཤེགས་པས་ཅི་ལ་དགོངས་ཏེ་དེ་བཞིན་གཤེགས་བའི་ཆོས་ཟབ་མོ་ཡང་དང་ཡང་གསུང་བ་མཛད་པ་དེ་བཅོམ་ལྡན་འདས་ཀྱིས་ལེགས་པར་བཤད་དུ་གསོལ་༑༑

ཡང་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་བཅོམ་ལྡན་འདས་ལ་ལན་གཉིས་སུ་གསོལ་བ་བཏབ་པ་༑་དོན་འདི་ཉིད་བཅོམ་ལྡན་འདས་བཤད་དུ་གསོལ་༑༑བདེ་བར་གཤེགས་པ་བཤད་དུ་གསོལ་༑་དེ་ཅིའི་སླད་དུ་ཞེ་ན་༑་བཅོམ་ལྡན ༼༢༽ འདས་འཁོར་འདི་ན་སྲོག་ཆགས་བརྒྱ་ཕྲག་མང་པོ་༑་སྲོག་ཆགས་སྟོང་ཕྲག་མང་པོ་༑་སྲོག་ཆགས་བརྒྱ་སྟོང་ཕྲག་མང་པོ་༑་སྲོག་ཆགས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་ཕྲག་མང་པོ་སྔོན་གྱི་སངས་རྒྱས་མཐོང་བ་ཤེས་རབ་དང་ལྡན་པ་གང་དང་༑༑་བཅོམ་ལྡན་འདས་ཀྱིས་བཤད་པ་ལ་དང ༼༣༽ པ་དང་ཡིད་ཆེས་པ་དང་འཛིན་པར་འགྱུར་བ་མཆིས་སོ་༑༑
དེ་ནས་དེའི་ཚེ་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་ཚིགས་སུ་བཅད་པ་འདི་གསོལ་ཏོ་༑༑
སྲོག་ཆགས་སྟོང་རྣམས་འཁོར་ན་མཆིས་པ་དག་༑༑་དང་དགའ་བདེ་བར་གཤེགས་ལ་གུས་པ་སྟེ་༑༑ ཁྱོད་ཀྱིས་ཆོས་བསྟན་དེ་དག་འཚལ་བར ༼༤༽ འགྱུར་༑༑་མི་དབང་རྒྱལ་པོ་གསལ་བར་བཤད་དུ་གསོལ་༑༑
དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་ལ་ལན་གཉིས་སུ་ཡང་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ་༑༑་ཤཱ་རིའི་བུ་དོན་འདི་བཤད་པས་ཅི་ཞིག་བྱ་༑་ཤཱ་རིའི་བུ་དོན་འདི་བཤད་ན་ལྷ་དང་བཅས་པའི་འཇིག་རྟེན ༼༥༽ འདི་དངངས་པར་འགྱུར་ཏེ་༑༑་དགེ་སློང་ལྷག་པའི་ང་རྒྱལ་ཅན་དུ་གྱུར་པ་རྣམས་ག་ཡངས་ཆེན་པོར་ལྷུང་བར་འགྱུར་རོ་༑༑
དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་དེའི་ཚེ་ཚིགས་སུ་བཅད་པ་འདི་དག་བཀའ་སྩལ་ཏོ་༑༑
ང་ཡི་ཆོས་འདི་བཤད་པས་ཆོག་༑༑་ཡེ་ཤེས་འདི་ཕྲ་བརྟག་མི་ནུས་༑༑ ༼༦༽ བྱིས་པ་མང་འཁོད་ང་རྒྱལ་ཅན་༑༑་ང་ཡིས་ཆོས་བཤད་མི་ཤེས་སྤོང་༑༑
དེ་ནས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་བཅོམ་ལྡན་འདས་ལ་ལན་གསུམ་དུ་ཡང་གསོལ་བ་བཏབ་པ་དོན་འདི་ཉིད་༑་བཅོམ་ལྡན་འདས་བཤད་དུ་གསོལ་༑་བདེ་བར་གཤེགས་པ་བཤད་དུ་གསོལ་༑༑བཅོམ ༼༧༽ ལྡན་འདས་འཁོར་འདི་ན་བདག་དང་འདྲ་བ་བརྒྱ་ཕྲག་མང་དུ་མཆིས་སོ་༑༑་བཅོམ་ལྡན་འདས་སྲོག་ཆགས་བརྒྱ་ཕྲག་མང་པོ་སྲོག་ཆགས་སྟོང་ཕྲག་མང་པོ་སྲོག་ཆགས་བརྒྱ་སྟོང་ཕྲག་མང་པོ་སྲོག་ཆགས་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་ཕྲག་མང་པོ་དག་བཅོམ་ལྡན་འདས་སྔོན་གྱི་སྲིད་པ ༼༡༦ཨ༽ ༼༡༦ཨ༡༽ ༑་དག་ན་ཡོངས་སུ་སྨིན་པར་མཛད་པ་གཞན་དག་ཀྱང་མཆིས་ཏེ་༑་བཅོམ་ལྡན་འདས་ཀྱིས་བཤད་པ་ལ་དད་པར་འགྱུར་ཡིད་ཆེས་པར་འགྱུར་འཛིན་པར་འགྱུར་རོ་༑༑་དེ་ཡང་དེ་དག་གི་ཡུན་རིང་པོའི་དོན་དང་སྨན་པ་དང་བདེ་བར་འགྱུར་རོ་༑༑

རྐང་གཉིས་རྣམས་ཀྱི་མཆོག་གིས་ཆོས་བཤད་གསོལ་༑༑་ཁྱོད་ལ་སྲས་ཀྱི་ཐུ་བོ་བདག་གསོལ་པ་༑༑ འདི་ན་སྲོག་ཆགས་བྱེ་བ་སྟོང་མཆིས་པ་༑༑་དེ་དག་ཁྱོད་ཀྱིས་ཆོས་བཤད་འདི་དང་འགྱུར་༑༑
སྔོན་གྱི་སྲིད་པ་རྣམས་སུ་ཁྱོད་ཀྱིས ༼༣༽ གང་༑༑་ཡུན་རིང་རྟག་ཏུ་སེམས་ཅན་སྨིན་མཛད་པ་༑༑ དེ་ཀུན་ཐལ་མོ་སྦྱར་ཏེ་འདི་ན་མཆིས་༑༑་དེ་དག་ཆོས་འདི་ལ་ནི་དང་པར་འགྱུར་༑༑
བདག་འདྲ་བརྒྱ་ཕྲག་བཅུ་གཉིས་འདི་དག་དང་༑༑་གང་དག་བྱང་ཆུབ་མཆོག་ཏུ་ཞུགས་པ་དག་༑༑ དེ་ལ་གཟིགས་ནས་བདེ་གཤེགས་རབ་བཤད་དེ་༑༑ ༼༤༽ དེ་དག་དགའ་བ་མཆོག་ཀྱང་བསྐྱེད་དུ་གསོལ་༑༑
དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུས་ལན་གསུམ་གསོལ་བ་བཏབ་པ་མཁྱེན་ནས་༑་ཚེ་དང་ལྡན་པ་ཤཱ་རིའི་བུ་ལ་འདི་སྐད་ཅེས་བཀའ་སྩལ་ཏོ་༑༑ཤཱ་རིའི་བུ་འདི་ལྟར་དེ་བཞིན་གཤེགས་པ་ལ་ལན་གསུམ་གྱི་བར ༼༥༽ དུ་གསོལ་བ་འདེབས་ན་༑་ཤཱ་རིའི་བུ་དེ་ལྟར་གསོལ་བ་འདེབས་པ་ལ་ཁྱོད་ལ་ཇི་སྐད་བྱས་༑ཤཱ་རིའི་བུ་དེའི་ཕྱིར་ལེགས་པར་རབ་ཏུ་ཉོན་ལ་ཡིད་ལ་ཟུང་ཤིག་དང་ངས་ཁྱོད་ལ་བཤད་དོ་༑༑
བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་སྩལ་མ་ཐག་ཏུ་༑་འཁོར་དེ་ནས་དགེ་སློང་དང ༼༦༽ དགེ་སློང་མ་དང་དགེ་བསྙེན་དང་དགེ་བསྙེན་མ་ལྷག་པའི་ང་རྒྱལ་ཅན་ལྔ་སྟོང་སྟན་ལས་ལངས་ཏེ་༑་བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ནས་འཁོར་དེ་ནས་དོང་ངོ་༑༑འདི་ལྟར་ལྷག་པའི་ང་རྒྱལ་གྱིས་དགེ་བའི་རྩ་བ་མ་ཐོབ་པར་ནི་ཐོབ་པར་འདུ་ཤེས་༑༑ཁོང་དུ་མ ༼༧༽ མ་ཆུད་པར་ནི་ཁོང་དུ་ཆུད་པར་འདུ་ཤེས་པ་དེ་དག་རང་གི་སྐྱོན་ཤེས་ཏེ་༑་འཁོར་དེ་ནས་དོང་བ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཅད་མི་གསུང་བར་གྱུར་པས་གནང་ངོ་༑༑
